#include <stdio.h>
#include <string.h>

// Variables globales
int inventario = 300;  // huevos disponibles
char clientes[10][50]; // lista de clientes
int totalClientes = 0;

// --- Registro de clientes ---
void registrarCliente() {
    if (totalClientes < 10) {
        printf("\nIngrese nombre del cliente: ");
        fflush(stdin);
        fgets(clientes[totalClientes], 50, stdin);

        // Eliminar salto de línea
        clientes[totalClientes][strcspn(clientes[totalClientes], "\n")] = 0;

        totalClientes++;
        printf("Cliente registrado exitosamente.\n");
    } else {
        printf("Limite de clientes alcanzado.\n");
    }
}

// --- Mostrar clientes ---
void mostrarClientes() {
    printf("\n Lista de Clientes \n");
    for (int i = 0; i < totalClientes; i++) {
        printf("%d. %s\n", i + 1, clientes[i]);
    }
    if (totalClientes == 0) {
        printf("No hay clientes registrados.\n");
    }
}

// --- Registrar venta ---
void registrarVenta() {
    int cantidad;

    printf("\nIngrese cantidad de huevos a vender: ");
    scanf("%d", &cantidad);

    if (cantidad <= inventario && cantidad > 0) {
        inventario -= cantidad;
        printf("Venta realizada con exito.\n");
        printf("Huevos restantes: %d\n", inventario);
    } else {
        printf("No hay suficiente inventario.\n");
    }
}

// --- Mostrar inventario ---
void mostrarInventario() {
    printf("\nHuevos disponibles: %d\n", inventario);
}

// --- Menú principal ---
int main() {
    int opcion;

    do {
        printf(" SISTEMA DE GESTION - DISTRIBUIDORA \n");
        printf("1. Registrar cliente\n");
        printf("2. Ver clientes\n");
        printf("3. Registrar venta\n");
        printf("4. Ver inventario\n");
        printf("5. Salir\n");
        printf("Seleccione una opcion: ");
        scanf("%d", &opcion);
        getchar(); //

        switch (opcion) {
            case 1:
                registrarCliente();
                break;
            case 2:
                mostrarClientes();
                break;
            case 3:
                registrarVenta();
                break;
            case 4:
                mostrarInventario();
                break;
            case 5:
                printf("Saliendo del sistema...\n");
                break;
            default:
                printf("Opcion no valida.\n");
        }

    } while (opcion != 5);

    return 0;
}
